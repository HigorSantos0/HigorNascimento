void main()
{
	int i, base, expoente, valor = 1;
	
	printf ("Entre com o valor da base:\n");
	scanf ("%d", &base);
	
	printf ("Entre com o valor do expoente:\n");
	scanf ("%d", &expoente);
	
	for (i = 1; i <= expoente; i++)
	{
		valor = valor * base;
		printf ("%d ", valor);
	}
	
	printf ("O resulta e' = %d\n", valor);

}
